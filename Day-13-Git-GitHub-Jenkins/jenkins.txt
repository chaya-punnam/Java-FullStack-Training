pipeline {
agent any

stages {

    stage('Clean Workspace') {
        steps {
            deleteDir()
        }
    }

    stage('Cloning Projects from GitHub') {
        steps {

            dir('doctor') {
                git url: 'https://github.com/chaya-punnam/Doctor-microservice.git', branch: 'master'
            }

            dir('hospital') {
                git url: 'https://github.com/chaya-punnam/Hospital-microservice.git', branch: 'master'
            }

            dir('Eureka') {
                git url: 'https://github.com/chaya-punnam/Netflix-Eureka-server.git', branch: 'master'
            }

            dir('APIGateWay') {
                git url: 'https://github.com/chaya-punnam/APIGateWay-microservice.git', branch: 'master'
            }
        }
    }

    stage('Creating JAR Files') {
        steps {

            dir('doctor') {
                bat 'mvn clean package -DskipTests'
            }

            dir('hospital') {
                bat 'mvn clean package -DskipTests'
            }

            dir('Eureka') {
                bat 'mvn clean package -DskipTests'
            }

            dir('APIGateWay') {
                bat 'mvn clean package -DskipTests'
            }
        }
    }

    stage('Running JAR Files') {
        steps {

            bat 'start cmd /c java -jar Eureka\\target\\NetflixEurekaServerProject-0.0.1-SNAPSHOT.jar --server.port=8033'

            bat 'start cmd /c java -jar hospital\\target\\HospitalApp-0.0.1-SNAPSHOT.jar --server.port=8032'

            bat 'start cmd /c java -jar doctor\\target\\DoctorApp-0.0.1-SNAPSHOT.jar --server.port=8031'

            bat 'start cmd /c java -jar APIGateWay\\target\\APIGateWay-0.0.1-SNAPSHOT.jar --server.port=8034'
        }
    }
}


}
